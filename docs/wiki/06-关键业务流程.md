// DeduplicationPostProcessor.java
public List<Chunk> process(List<Chunk> chunks) {
    Map<String, Chunk> uniqueChunks = new HashMap<>();
    
    for (Chunk chunk : chunks) {
        String key = chunk.getChunkId();
        
        if (!uniqueChunks.containsKey(key)) {
            uniqueChunks.put(key, chunk);
        } else {
            // 合并分数（混合检索融合）

            double fusedScore = 0.7 * existing.getScore() + 0.3 * chunk.getScore();
            existing.setScore(fusedScore);
        }
    }
    
    return new ArrayList<>(uniqueChunks.values());
}
